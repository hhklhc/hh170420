Câu 10: Tính số lượng nhân viên theo từng mã phòng ban


SELECT Department, COUNT(SSN)
FROM Employees
GROUP BY Department

Câu 8 Tính tổng ngân quỹ cho tất cả các phòng ban


SELECT SUM(Budget)
FROM Departments

Câu 9 Tính trung bình ngân quỹ cho tất cả các phòng ban


SELECT AVG(Budget)

Câu 7 Lấy tất cả thông tin của các nhân viên có last name kết thúc bằng chữ "th" hoặc "on", đồng thời có name chứa chữ cái 'a'


SELECT *
FROM Employees
WHERE Name LIKE '%a%' AND ( LastName LIKE '%th' OR LastName LIKE '%on')

Câu 6 Lấy tất cả thông tin của các nhân viên có last name bắt đầu bằng chữ cái "S" hoặc "D"


SELECT *
FROM Employees
WHERE LastName LIKE 'S%' OR LastName LIKE 'D%'

Câu 5 Lấy tất cả thông tin của các nhân viên làm việc ở phòng số 14, 37 hoặc 77


SELECT *
FROM Employees
WHERE Department IN ('14', '37', '77')

SELECT *

Câu 3 Lấy tất cả thông tin của các nhân viên có last name là "Smith"
